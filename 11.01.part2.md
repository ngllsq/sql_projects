UNION
SELECT id, name FROM person

#Task - 2
```sql
SELECT object_name FROM (
	SELECT pizza_name AS object_name, 'pizza' AS type FROM menu
	UNION ALL
	SELECT name, 'name' FROM person
	ORDER BY TYPE DESC
) AS tablee;
```
